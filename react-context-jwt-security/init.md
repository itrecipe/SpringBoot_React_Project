# React 앱 생성

npx create-react-app client

# axios (비동기 통신을 위해 axios 라이브러리를 설치하여 서버와 통신을 수행한다.)

npm install axios

# router

npm install react-router-dom

# cookie (리액트에서 엑시오스로)

npm install js-cookie

# sweetalert2

npm install sweetalert2
npm install sweetalert2-react-content
